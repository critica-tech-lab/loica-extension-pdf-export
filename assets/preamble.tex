% ── Loica PDF preamble ─────────────────────────────────────
% Matches iA Writer "Modern (Sans)" template print typography.
% Font: IBM Plex Sans / Mono.  Base: 11pt, line-height 1.55.

% ── Base font size override: 10.5pt to match iA Writer print ─
% Calibrated visually against iA Writer PDF output in 365pt column.
\renewcommand{\normalsize}{\fontsize{10.5pt}{16.28pt}\selectfont}
\normalsize

% ── Ragged-right, no hyphenation (iA Writer print style) ─
\usepackage{ragged2e}
\RaggedRight
\hyphenpenalty=10000
\exhyphenpenalty=10000

% ── Microtypography ───────────────────────────────────────
\usepackage{microtype}

% ── Colors ────────────────────────────────────────────────
\usepackage{xcolor}
\definecolor{body}{HTML}{1A1A1A}
\definecolor{linkblue}{HTML}{3366CC}
\definecolor{codebg}{HTML}{F4F4F4}
\definecolor{codefg}{HTML}{1A1A1A}
\definecolor{rulegray}{HTML}{1A1A1A}
\definecolor{quotegray}{HTML}{1A1A1A}

\color{body}

% ── Utility ──────────────────────────────────────────────
\usepackage{etoolbox}

% ── Page geometry ─────────────────────────────────────────
% iA Writer print: body padding 12pt 50pt, max-width 365pt.
% On A4/Letter the text block is ~365pt centered with generous
% side margins. Top/bottom ~50pt to match the horizontal feel.
\usepackage[
  top=50pt,
  bottom=60pt,
  left=\dimexpr(\paperwidth - 365pt) / 2\relax,
  right=\dimexpr(\paperwidth - 365pt) / 2\relax,
]{geometry}

% ── Link styling ──────────────────────────────────────────
% hyperref is loaded by pandoc; link colors are passed via
% -V colorlinks/urlcolor/linkcolor from the PDF route so that
% pandoc's default \hypersetup emits a colored setup instead of
% `hidelinks`. We only define the color names here.

% ── Float placement ──────────────────────────────────────
\usepackage{float}
\floatplacement{figure}{H}

% ── Graphics ─────────────────────────────────────────────
% Pandoc loads graphicx only when the body has images. Cover-page templates
% (report mode) include the org wordmark via \includegraphics, so load it here.
\usepackage{graphicx}

% ── Tables ───────────────────────────────────────────────
% iA: top+bottom border only, no vertical rules, 0.8333em font, full width
\usepackage{booktabs}
\usepackage{longtable}
\setlength{\heavyrulewidth}{0.4pt}
\setlength{\lightrulewidth}{0.4pt}
\setlength{\aboverulesep}{0.6ex}
\setlength{\belowrulesep}{0.6ex}
\renewcommand{\arraystretch}{1.4}

% Force longtable to span the full text width
\setlength{\LTleft}{0pt}
\setlength{\LTright}{0pt}
\AtBeginEnvironment{longtable}{\small}

% ── Headings ─────────────────────────────────────────────
% iA Writer Sans heading scale (relative to 18px base):
%   H1: 1.2778em (23/18), line-height 1.304, bold
%   H2: 1.1667em (21/18), line-height 1.4286, bold
%   H3–H6: 1em, bold (H5: medium weight)
\usepackage{titlesec}

% H1 — 1.2778em = 13.42pt, line-height 1.304 = 17.5pt
\titleformat{\section}
  {\fontsize{13.42pt}{17.5pt}\selectfont\bfseries}
  {}{0em}{}

% H2 — 13pt bold (top-level section in body without H1)
\titleformat{\subsection}
  {\fontsize{13pt}{18pt}\selectfont\bfseries}
  {}{0em}{}

% H3 — 12pt bold + small caps (ranks clearly above 10.5pt inline bold)
\titleformat{\subsubsection}
  {\fontsize{12pt}{17pt}\selectfont\bfseries\scshape}
  {}{0em}{}

% H4 — 1em, bold italic (distinct from H3 and from inline bold)
\titleformat{\paragraph}
  {\normalsize\bfseries\itshape}
  {}{0em}{}

% H5 — 1em, medium weight
\titleformat{\subparagraph}
  {\normalsize\fontseries{m}\selectfont}
  {}{0em}{}

% iA print spacing (at 9pt base):
%   H1: 60px→30pt top+bottom, H2: 60px→30pt top / 30px→15pt bottom
%   H3-H6: 30px→15pt top, 0 bottom
\titlespacing*{\section}      {0pt}{35pt plus 4pt minus 2pt}{35pt plus 2pt minus 1pt}
\titlespacing*{\subsection}   {0pt}{35pt plus 4pt minus 2pt}{17.5pt plus 2pt minus 1pt}
\titlespacing*{\subsubsection}{0pt}{24pt plus 3pt minus 1pt}{4pt}
\titlespacing*{\paragraph}    {0pt}{17.5pt plus 3pt minus 1pt}{0pt}
\titlespacing*{\subparagraph} {0pt}{17.5pt plus 3pt minus 1pt}{0pt}

% ── Code blocks ──────────────────────────────────────────
% iA: very light gray bg, 0.94em font
\usepackage{framed}

\definecolor{shadecolor}{HTML}{F0F0F0}
\AfterEndPreamble{%
  \ifdefined\Shaded
    \renewenvironment{Shaded}{%
      \vspace{0.5ex}%
      \begin{snugshade}%
      \small\color{codefg}%
    }{%
      \end{snugshade}%
      \vspace{0.5ex}%
    }%
  \fi
}

% Inline code: 0.94em, subtle background
\let\oldtexttt\texttt
\renewcommand{\texttt}[1]{%
  \colorbox{codebg}{\oldtexttt{\fontsize{9.87pt}{10.5pt}\selectfont\color{codefg}#1}}%
}

% ── Block quotes ─────────────────────────────────────────
% iA: no visible border, just indented left and right
\usepackage{mdframed}
\renewenvironment{quote}{%
  \begin{mdframed}[
    topline=false,
    rightline=false,
    bottomline=false,
    leftline=false,
    innerleftmargin=2.2em,
    innerrightmargin=2.2em,
    innertopmargin=0pt,
    innerbottommargin=0pt,
    skipabove=17.5pt,
    skipbelow=0pt,
  ]%
}{%
  \end{mdframed}%
}

% ── Suppress default title ───────────────────────────────
\renewcommand{\maketitle}{}

% ── Lists ────────────────────────────────────────────────
% iA Writer print: justified text, tight bullets, ~2.2em indent
\usepackage{enumitem}
\setlist{nosep, leftmargin=1.5em, topsep=0pt, partopsep=0pt, parsep=0pt, itemsep=0pt}
\setlist[itemize,1]{leftmargin=1.5em, label=\textbullet}
\setlist[itemize,2]{label=\textendash}
\setlist[enumerate,1]{leftmargin=2.2em}

% ── Paragraph spacing ───────────────────────────────────
% iA: no indent, block elements have margin-top: 1.667em, margin-bottom: 0
\setlength{\parindent}{0pt}
\setlength{\parskip}{17.5pt plus 2pt minus 1pt}

% ── Horizontal rule ──────────────────────────────────────
% Handled by Lua filter (full-width line)

% ── Unicode geometric arrows fallback ───────────────────
% IBM Plex Sans lacks U+25B2/U+25BC; substitute with arrows it does have
\usepackage{newunicodechar}
\newunicodechar{▲}{↑}
\newunicodechar{▼}{↓}

% ── Table of contents formatting ─────────────────────────
% In docs without H1, md-H2 (\subsection) is the top-level section.
% Render H2 entries flush-left + bold; H3 entries indented + lighter.
\usepackage{titletoc}
\dottedcontents{subsection}[0pt]{\addvspace{0.6em}\bfseries}{0pt}{0.5em}
\dottedcontents{subsubsection}[1.5em]{\footnotesize\color{black!70}}{0pt}{0.4em}

% ── Source notes (table/figure attributions) ────────────
% Lua filter wraps any paragraph beginning with "Source:" in this env
\newenvironment{sourcenote}{%
  \par\vspace{-0.5\baselineskip}%
  \footnotesize\color{black!55}%
}{\par\vspace{0.5\baselineskip}}

% ── Footnotes ────────────────────────────────────────────
% flushmargin: marker sits at the left margin (no first-line indent)
% hang: continuation lines align under the text, not under the marker
\usepackage[hang,flushmargin]{footmisc}
\setlength{\footnotemargin}{0.6em}

% ── Page numbers: centered, small ────────────────────────
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
\fancyfoot[C]{\footnotesize\thepage}
\setlength{\footskip}{1.5cm}
